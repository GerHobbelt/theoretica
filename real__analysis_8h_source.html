58 template<typename UnsignedIntType = uint64_t>
59 inline UnsignedIntType isqrt(UnsignedIntType n) {
60
61 // Upper bound
62 UnsignedIntType upper = n + 1;
63
64 // Lower bound
65 UnsignedIntType lower = 0;
66
67 // Carry for safe long division
68 UnsignedIntType c = 0;
69
70 while(lower != upper - 1) {
71
72 // Compute carry for long division by 2
73 c = ((lower % 2 != 0) && (upper % 2 != 0)) ? 1 : 0;
74
75 // Safer division by 2 for big numbers
76 const UnsignedIntType m = (lower >> 1) + (upper >> 1) + c;
77
78 // Using division instead of multiplication avoids
79 // overflows which would remove significant bits
80 const UnsignedIntType q = n / m;
81
82 if(m > q)
83 upper = m;
84 else if(m < q)
85 lower = m;
86 else
87 return m;
88 }
89
90 return lower;
91 }

203 inline real cbrt(real x) {
204
205 if(x < 1) {
206
207 if(x == 0)
208 return 0;
209
210 // cbrt(x) is odd
211 if(x < 0)
212 return -cbrt(-x);
213
214 // Approximate cbrt between 0 and 1
215 // The root of the inverse is the inverse of the root
216 // !!! Possible precision problems with smaller numbers
217 return 1.0 / cbrt(1.0 / x);
218 }
219
220 // Approximate cbrt(x) using Newton-Raphson
221 real y = x;
222 unsigned int i = 0;
223
224 while((cube(y) - x) > OPTIMIZATION_TOL && i < OPTIMIZATION_NEWTON_ITER) {
225 y = (y * 2.0 + x / (y * y)) / 3.0;
226 i++;
227 }
228
229 return y;

429 inline real log2(real x) {
430
431 if(x <= 0) {
432
433 if(x == 0) {
434 TH_MATH_ERROR("log2", x, OUT_OF_RANGE);
435 return -inf();
436 }
437
438 TH_MATH_ERROR("log2", x, OUT_OF_DOMAIN);
439 return nan();
440 }
441
442#ifdef THEORETICA_X86
443
444 // Approximate the binary logarithm of x by
445 // exploiting x86 Assembly instructions
446 return fyl2x(x, 1.0);
447#else
448
449 // Domain reduction to [1, +inf)
450 if (x < 1.0)
451 return -log2(1.0 / x);
452
453 // Compute the biggest power of 2 so that x <= 2^i
454 unsigned int i = 0;
455 while(x > (uint64_t(1) << i))
456 i++;
457
458 // Domain reduction to [1, 2]
459 x /= (uint64_t(1) << i);
460
461 // Use the Taylor expansion of the logarithm
462 // ln(1 + x) = \sum_k^n (-1)^(k+1) x^k / k
463 real log_z = 0;
464 const real z = x - 1;
465
466 // Exact powers of 2 don't need further computation
467 if(abs(z) > MACH_EPSILON) {
468
469 int sign = 1;
470 real pow_z = z;
471 log_z = z;
472
473 for (int i = 2; i <= 24; ++i) {
474
475 pow_z *= z;
476 sign *= -1;
477
478 log_z += sign * pow_z / i;
479 }
480 }
481
482 return i + (log_z / LN2);
483#endif
484 }

606 template<typename T = real>
607 TH_CONSTEXPR inline T pow(T x, int n) {
608
609 if(n > 0) {
610
611 T res = x;
612 T x_sqr = x * x;
613 int i = 1;
614
615 // Self-multiply up to biggest power of 2
616 for (; i < (n / 2); i *= 2)
617 res = res * res;
618
619 // Multiply by x^2 for remaining even powers
620 for (; i < (n - 1); i += 2)
621 res = res * x_sqr;
622
623 // Multiply for remaining powers
624 for (; i < n; ++i)
625 res = res * x;
626
627 return res;
628
629 } else if(n < 0) {
630 return T(1.0) / pow(x, -n);
631 } else {
632 return 1;
633 }
634 }

742 inline real exp(real x) {
743
744 // Domain reduction to [0, +inf]
745 if(x < 0)
746 return 1.0 / exp(-x);
747
748 const real fract_x = fract(x);
749 const int floor_x = floor(x);
750
751 // Taylor series expansion
752 // Compute e^floor(x) * e^fract(x)
753
754 real res = 1;
755 real s_n = 1;
756
757 for (int i = 1; i <= CORE_TAYLOR_ORDER; ++i) {
758
759 // Recurrence formula to improve
760 // numerical stability and performance
761 s_n *= fract_x / (i * 4);
762 res += s_n;
763 }
764
765 // The fractional part is divided by 4 to improve convergence
766 const real sqr_r = res * res;
767 return pow(E, floor_x) * sqr_r * sqr_r;
768 }

816 inline real root(real x, int n) {
817
818 if(((n % 2 == 0) && (x < 0)) || (n == 0)) {
819 TH_MATH_ERROR("root", n, OUT_OF_DOMAIN);
820 return nan();
821 }
822
823 if(n < 0)
824 return 1.0 / root(x, -n);
825
826 // Trivial cases
827 if(n == 1)
828 return x;
829
830 if(n == 2)
831 return sqrt(x);
832
833 if(n == 3)
834 return cbrt(x);
835
836 if(x < 1) {
837
838 if(x == 0)
839 return 0;
840
841 // Approximate root between 0 and 1
842 // The root of the inverse is the inverse of the root
843 // !!! Possible precision problems with smaller numbers
844 return 1.0 / root(1.0 / x, n);
845 }
846
847 // Approximate n-th root using Newton-Raphson.
848 // If fast exponentials and logarithms are available,
849 // use a first calculation to speed up convergence.
850#ifdef THEORETICA_X86
851 real y = exp(ln(x) / n);
852#else
853 real y = x;
854#endif
855
856 unsigned int i = 0;
857
858 while(i < OPTIMIZATION_NEWTON_ITER) {
859
860 const real y_pow = pow(y, n - 1);
861
862 if((y_pow * y - x) < OPTIMIZATION_TOL)
863 break;
864
865 y = (y * (n - 1) + x / y_pow) / (real) n;
866 i++;
867 }
868
869 if(i >= OPTIMIZATION_NEWTON_ITER) {
870 TH_MATH_ERROR("root", i, NO_ALGO_CONVERGENCE);
871 return nan();
872 }
873
874 return y;
875 }
